  -DarchetypeArtifactId=maven-archetype-quickstart \

  -DgroupId=com.github.saviosailas \
  -DartifactId="MyGame" \
  -Dversion=1.0-SNAPSHOT \
  -Dpackage=com.gitlab.saviosailas.my_game \
  -DinteractiveMode=false
